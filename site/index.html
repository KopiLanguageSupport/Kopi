// Define a function to brew coffee
func brewCoffee(type: String, shots: Number) -> String {
  if shots > 0 {
    return "Brewing a hot " + type + " with " + shots + " shots!"
  }
  return "Just hot water for now."
}

print(brewCoffee(type: "Espresso", shots: 2))
// Output: Brewing a hot Espresso with 2 shots!
